  • 入れ子（ネスト）ができる

①入れ子（ネスト）ができる

nav ul { margin: 0; }
nav li { display: inline-block; }
↓ 
nav {
  ul { margin: 0; }
  li { display: inline-block; }
}

②変数が使える

$primary-color: #333;

body {
  color: $primary-color;
}
↓ 
body { color: #333; }

③演算子が使える

article {
  width: 600px / 960px * 100%;
}
↓ 
article {
  width: 62.5%;
}

④セレクタが使いまわせる

.button {
    display: inline-block;
　　padding:20px;
　　border-radius: 30px;}
  
.button--red {
　　@extend .button;
　　background: red;
}

⑤コードが再利用できる

@mixin button{
　　display: inline-block;
　　padding: 10px;
}

.button--red {
　　@include button;
    background: red;
}

⑥コードが再利用できる（引数を使う）

　padding: 10px;
　background: $color;
}

.button--normal {@include button;}

.button--red {@include button(red);}

// gulpプラグインの読み込み
const gulp = require('gulp');
// Sassをコンパイルするプラグインの読み込み
const sass = require('gulp-sass');
 
// style.scssをタスクを作成する
gulp.task('default', function () {
  // style.scssファイルを取得
  gulp.src('css/style.scss')
    // Sassのコンパイルを実行
    .pipe(sass({
      outputStyle: 'expanded'
    }))
    // cssフォルダー以下に保存
    .pipe(gulp.dest('css'));
});

Gulpを使った自動処理の設定

watch機能を使ってSassファイルを自動コンパイルしてみよう

gulpfile.jsを用意する

// gulpプラグインの読み込み
const gulp = require('gulp');
// Sassをコンパイルするプラグインの読み込み
const sass = require('gulp-sass');
  
// style.scssの監視タスクを作成する
gulp.task('default', function () {
  // ★ style.scssファイルを監視
  gulp.watch('css/style.scss', function () {
    // style.scssの更新があった場合の処理
  
    // style.scssファイルを取得
    gulp.src('css/style.scss')
      // Sassのコンパイルを実行
      .pipe(sass({
        outputStyle: 'expanded'
      })
      // Sassのコンパイルエラーを表示
      // (これがないと自動的に止まってしまう)
      .on('error', sass.logError))
      // cssフォルダー以下に保存
      .pipe(gulp.dest('css'));
  });
});

npx gulp

watchを終了する
